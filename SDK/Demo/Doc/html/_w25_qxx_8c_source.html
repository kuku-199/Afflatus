189uint8_t W25Qx_Write_Data(uint8_t *Data, uint32_t WriteAddr, uint32_t Size)
190{
191 uint8_t cmd[4];
192 uint32_t end_addr, current_size, current_addr;
193 uint32_t StartTime = HAL_GetTick();
194
195 /* Calculation of the size between the write address and the end of the page */
196 current_addr = 0;
197
198 while (current_addr <= WriteAddr)
199 {
200 current_addr += W25QX_PAGE_SIZE;
201 }
202 current_size = current_addr - WriteAddr;
203
204 /* Check if the size of the data is less than the remaining place in the page */
205 if (current_size > Size)
206 {
207 current_size = Size;
208 }
209
210 /* Initialize the adress variables */
211 current_addr = WriteAddr;
212 end_addr = WriteAddr + Size;
213
214 /* Perform the write page by page */
215 do
216 {
217 /* Configure the command */
218 cmd[0] = FLASH_Write;
219 cmd[1] = (uint8_t)(current_addr >> 16);
220 cmd[2] = (uint8_t)(current_addr >> 8);
221 cmd[3] = (uint8_t)(current_addr);
222
223 /* Enable write operations */
224 W25Qx_WriteEnable();
225
226 Flash_CS_Select();
227 /* Send the command */
228 if (HAL_SPI_Transmit(&hspi2, cmd, 4, W25Qx_TIMEOUT_VALUE) != HAL_OK)
229 {
230 return W25Qx_ERROR;
231 }
232
233 /* Transmission of the data */
234 if (HAL_SPI_Transmit(&hspi2, Data, current_size, W25Qx_TIMEOUT_VALUE) != HAL_OK)
235 {
236 return W25Qx_ERROR;
237 }
238 Flash_CS_Discard();
239 /* Wait the end of Flash writing */
240 while (W25Qx_Read_Busy() == W25Qx_BUSY)

244 {
245 return W25Qx_TimeOut;
246 }
247 // delay(1);
248 }
249
250 /* Update the address and size variables for next page programming */
251 current_addr += current_size;
252 Data += current_size;
253 current_size = ((current_addr + W25QX_PAGE_SIZE) > end_addr) ? (end_addr - current_addr) : W25QX_PAGE_SIZE;
254 } while (current_addr < end_addr);
